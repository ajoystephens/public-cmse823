# calculate p(x) using coefficients per formula in (a)
def getP_a(x):
    p=x**9
    p-=18*(x**8)
    p+=144*(x**7)
    p-=672*(x**6)
    p+=2016*(x**5)
    p-=4032*(x**4)
    p+=5376*(x**3)
    p-=4608*(x**2)
    p+=2304*x
    p-=512
    return(p)

# calculate p(x) using (x-2)^9 per part (b)
def getP_b(x):
    p=(x-2)**9
    return(p)
